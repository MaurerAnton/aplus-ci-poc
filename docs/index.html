Quick Start

git clone https://github.com/MaurerAnton/aplus-ci-poc
cd aplus-ci-poc

# Build A+ interpreter:
docker build -t aplus:ci .

# Run a program:
echo '2+3' | docker run --rm -i aplus:ci

# Transpile to Python/JS/Go/Rust/C:
python3 transpile_aplus.py --input hello.a+ --output-py out.py --output-js out.js

# Play 20 Questions:
python3 play_20q.py

# Run benchmarks:
python3 benchmark.py /path/to/a+
